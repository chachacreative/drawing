<body>

    <h1>Imagination Prompt Game</h1>

    <div id="prompt">Loading prompt...</div>

    <div id="timer">30</div>

    <button id="startBtn" onclick="startGame()">Start</button>

    <button id="pauseBtn" onclick="pauseGame()" disabled>Pause</button>


    <script>

        let prompts = [

            "一個動物，頭像雞，身體像鳳梨，腳像鴨子的腳，在廣場上跳舞。",

            "一個外星人，把時間暫停，拿著放大鏡在觀察人類在做什麼。",

            "一張超級英雄，用有一個可以透視蔬菜的能力。",

            // Add more prompts here...

        ];


        let currentIndex = 0;

        let timer;

        let timeLeft = 30;


        function displayPrompt() {

            document.getElementById("prompt").textContent = prompts[currentIndex];

            timeLeft = 30;

            document.getElementById("timer").textContent = timeLeft;

        }


        function startGame() {

            displayPrompt();

            timer = setInterval(updateTimer, 1000);

            document.getElementById("startBtn").disabled = true;

            document.getElementById("pauseBtn").disabled = false;

        }


        function pauseGame() {

            clearInterval(timer);

            document.getElementById("startBtn").disabled = false;

            document.getElementById("pauseBtn").disabled = true;

        }


        function updateTimer() {

            timeLeft--;

            document.getElementById("timer").textContent = timeLeft;

            if (timeLeft === 0) {

                clearInterval(timer);

                currentIndex++;

                if (currentIndex < prompts.length) {

                    displayPrompt();

                    timer = setInterval(updateTimer, 1000);

                } else {

                    document.getElementById("prompt").textContent = "遊戲結束！";

                    document.getElementById("timer").textContent = "";

                    document.getElementById("startBtn").disabled = true;

                    document.getElementById("pauseBtn").disabled = true;

                }

            }

        }

    </script>

</body>
